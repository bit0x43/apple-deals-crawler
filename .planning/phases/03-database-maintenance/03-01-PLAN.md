**src/apple_deals/db/crud.py** implementation:

```python
import os
from datetime import datetime, timedelta, timezone

from sqlalchemy import delete, func, select, text
from sqlalchemy.engine import Engine
from sqlalchemy.orm import Session

from apple_deals.db.models import Product


def _naive_utc_cutoff(retention_days: int) -> datetime:
    """Return a timezone-naive UTC cutoff datetime for the given retention window.

    Uses .replace(tzinfo=None) because Product.crawled_at is DateTime (not DateTime(timezone=True)).
    SQLite strips tzinfo on roundtrip; comparing aware vs naive raises TypeError.
    """
    return datetime.now(timezone.utc).replace(tzinfo=None) - timedelta(days=retention_days)


def prune_old_records(session: Session, retention_days: int = 90) -> int:
    """Delete Product rows older than retention_days. Returns count deleted."""
    cutoff = _naive_utc_cutoff(retention_days)
    stmt = delete(Product).where(Product.crawled_at < cutoff)
    result = session.execute(stmt)
    session.commit()
    return result.rowcount  # type: ignore[return-value]


def count_prunable(session: Session, retention_days: int = 90) -> int:
    """Count rows that would be deleted by prune_old_records. No side effects."""
    cutoff = _naive_utc_cutoff(retention_days)
    count = session.scalar(
        select(func.count()).select_from(Product).where(Product.crawled_at < cutoff)
    )
    return count or 0


def get_db_stats(session: Session, engine: Engine) -> dict:
    """Return DB statistics as a dict with keys:
    total_rows, oldest, newest, daily_growth, db_size.
    """
    total = session.scalar(select(func.count()).select_from(Product)) or 0
    oldest = session.scalar(select(func.min(Product.crawled_at)))
    newest = session.scalar(select(func.max(Product.crawled_at)))

    cutoff_7d = _naive_utc_cutoff(7)
    recent = session.scalar(
        select(func.count()).select_from(Product).where(Product.crawled_at >= cutoff_7d)
    ) or 0
    daily_growth = round(recent / 7, 1) if recent else 0.0

    is_postgres = engine.url.drivername.startswith("postgresql")
    if is_postgres:
        size_str = session.scalar(
            text("SELECT pg_size_pretty(pg_database_size(current_database()))")
        ) or "unknown"
    else:
        db_path = engine.url.database
        if db_path == ":memory:" or db_path is None:
            size_str = "in-memory"
        else:
            try:
                size_bytes = os.path.getsize(db_path)
                size_str = f"{size_bytes / 1024:.1f} KB"
            except FileNotFoundError:
                size_str = "unknown"

    return {
        "total_rows": total,
        "oldest": oldest,
        "newest": newest,
        "daily_growth": daily_growth,
        "db_size": size_str,
    }
```
