```toml
    "ruff>=0.15.12",
    "mypy>=2.0.0",
    "detect-secrets>=1.5.0",
    "pytest>=9.0.3",
    "sqlalchemy[mypy]>=2.0.49",
]

[tool.ruff]
src = ["src"]
line-length = 100
target-version = "py313"

[tool.ruff.lint]
select = ["E", "F", "I", "UP"]

[tool.mypy]
python_version = "3.13"
mypy_path = "src"
plugins = ["sqlalchemy.ext.mypy.plugin"]
ignore_missing_imports = true
disallow_untyped_defs = true
warn_return_any = true
warn_unused_ignores = true

[tool.pytest.ini_options]
pythonpath = ["src"]
testpaths = ["tests"]
```

Critical checks:

- `[tool.mypy] mypy_path = "src"` and `plugins = ["sqlalchemy.ext.mypy.plugin"]` MUST be present
- `[tool.pytest.ini_options] pythonpath = ["src"]` MUST be present (Pitfall 1 in RESEARCH.md)
