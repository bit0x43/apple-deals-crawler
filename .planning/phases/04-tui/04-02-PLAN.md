    ```python
    from __future__ import annotations

    from textual import work
    from textual.app import App, ComposeResult
    from textual.containers import Horizontal
    from textual.screen import Screen
    from textual.widgets import DataTable, Footer, Header, Label, Select
    from textual.worker import get_current_worker

    from apple_deals.db.crud import get_current_prices
    from apple_deals.db.session import get_session


    MODEL_CHOICES = [
        ("All", "all"),
        ("MacBook Air", "MacBook Air"),
        ("MacBook Pro", "MacBook Pro"),
        ("Mac Mini", "Mac Mini"),
        ("Mac Studio", "Mac Studio"),
    ]

    STORE_CHOICES = [
        ("All", "all"),
        ("Tiendasishop", "tiendasishop"),
        ("Mac Center", "mac-center"),
    ]


    class CatalogScreen(Screen):
        """Browse current prices with filterable DataTable and sort-by-price."""

        _sort_reverse: bool = False

        def compose(self) -> ComposeResult:
            yield Header(id="catalog-header")
            with Horizontal(classes="filters"):
                yield Select(
                    MODEL_CHOICES,
                    prompt="Model",
                    id="model-filter",
                    value="all",
                )
                yield Select(
                    STORE_CHOICES,
                    prompt="Store",
                    id="store-filter",
                    value="all",
                )
            yield DataTable(id="catalog-table")
            yield Footer()

        def on_mount(self) -> None:
            table = self.query_one("#catalog-table", DataTable)
            table.cursor_type = "row"
            table.zebra_stripes = True
            table.add_columns("Model", "Store", "Price (COP)", "Memory", "Storage", "Color")
            self.load_data()

        @work(thread=True, exclusive=True)
        async def load_data(self) -> None:
            """Fetch current prices in a thread worker to avoid blocking the UI."""
            worker = get_current_worker()
            model_val = self.query_one("#model-filter", Select).value
            store_val = self.query_one("#store-filter", Select).value
            model_filter = None if model_val == "all" else str(model_val)
            store_filter = None if store_val == "all" else str(store_val)

            session = get_session()
            try:
                rows = get_current_prices(session, model_filter=model_filter, store_filter=store_filter)
            finally:
                session.close()

            if not worker.is_cancelled:
                self.call_from_thread(self._populate_table, rows)

        def _populate_table(self, rows: list) -> None:
            """Populate the DataTable with price rows, then sort default by price ASC."""
            table = self.query_one("#catalog-table", DataTable)
            table.clear()
            for p in rows:
                table.add_row(
                    p.reference,
                    p.source,
                    f"{p.price:,.0f}",
                    p.memory or "\u2014",
                    p.storage or "\u2014",
                    p.color or "\u2014",
                    key=str(p.id),
                )
            table.sort("Price (COP)")

        def on_select_changed(self, event: Select.Changed) -> None:
            """Re-query data when either filter changes."""
            if event.select.id in ("model-filter", "store-filter"):
                self.load_data()

        def on_data_table_header_selected(self, event: DataTable.HeaderSelected) -> None:
            """Toggle sort on Price column header click."""
            if event.column_key.label and "Price" in str(event.column_key.label):
                self._sort_reverse = not self._sort_reverse
                self.query_one("#catalog-table", DataTable).sort(
                    event.column_key, reverse=self._sort_reverse
                )


    class HistoryScreen(Screen):
        """Stub history screen -- replaced by Plan 03."""

        def compose(self) -> ComposeResult:
            yield Label("History View")


    class AppleDealsApp(App):
        """Interactive TUI for browsing Apple Mac prices."""

        TITLE = "Apple Deals"

        SCREENS = {
            "catalog": CatalogScreen,
            "history": HistoryScreen,
        }

        BINDINGS = [
            ("q", "quit", "Quit"),
        ]

        def on_mount(self) -> None:
            self.push_screen("catalog")


    def run_tui() -> None:
        """Launch the interactive TUI. Blocks until the user quits."""
        app = AppleDealsApp()
        app.run()
    ```

    **IMPORTANT NOTES:**
    - Keep HistoryScreen as a minimal Label stub -- it is NOT replaced in this plan
    - The `on_select_changed` handler fires for ALL Select.Changed events. The guard `if event.select.id in (...)` ensures only the two filter widgets trigger re-queries.
    - DataTable `sort("Price (COP)")` uses the column label string as the key. DataTable generates ColumnKeys from column labels when no explicit key is provided.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Widget state -> DB query | Select values from filter widgetsflow into get_current_prices() model_filter/store_filter. Values are constrained to the predefined choice enums. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03 | Tampering | Select widget filter values | mitigate | Values are restricted to the MODEL_CHOICES / STORE_CHOICES enums defined in Python. User cannot enter arbitrary strings -- Select only emits defined values. |
| T-04-04 | Denial of Service | Thread worker re-queries on every filter change | accept | exclusive=True on the worker cancels any in-flight query before starting a new one. Fast filter changes do not accumulate. |
</threat_model>
